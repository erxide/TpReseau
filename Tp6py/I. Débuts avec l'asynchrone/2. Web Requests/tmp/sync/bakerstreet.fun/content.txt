            const loadingDots = document.getElementById('loading-dots');
            
            function animateDots() {
                loadingDots.innerHTML = '';
                
                for (let i = 0; i < 3; i++) {
                    setTimeout(() => {
                        loadingDots.innerHTML += '.';
                    }, i * 500);
                }

                
                setTimeout(() => {
                    loadingDots.innerHTML = '';
                    animateDots();  
                }, 2000);
            }

            setTimeout(() => {
                animateDots();
